```
@if (AbTesting::isExperiment('logo-big'))
    <div class="logo-big"></div>
@elseif (AbTesting::isExperiment('logo-grayscale'))
    <div class="logo-greyscale"></div>
@elseif (AbTesting::isExperiment('brand-name'))
    <h1>Brand name</h1>
@endif
```
